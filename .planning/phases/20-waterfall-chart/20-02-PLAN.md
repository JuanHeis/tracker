  <action>
    Create `hooks/useMonthlyFlowData.ts` -- a thin hook that wraps computeWaterfallData with useMemo for React reactivity (FLOW-04).

    Structure:
    ```typescript
    import { useMemo } from "react";
    import { computeWaterfallData } from "@/lib/projection/waterfall";
    import type { WaterfallBar } from "@/lib/projection/waterfall";
    import type { Expense, ExtraIncome, Investment } from "@/hooks/useMoneyTracker";
    import type { ViewMode } from "@/hooks/usePayPeriod";

    export function useMonthlyFlowData(
      expenses: Expense[],
      investments: Investment[],
      salaryAmount: number,
      extraIncomes: ExtraIncome[],
      selectedMonth: string,
      viewMode: ViewMode,
      payDay: number,
    ): WaterfallBar[] {
      return useMemo(
        () =>
          computeWaterfallData({
            expenses,
            investments,
            salaryAmount,
            extraIncomes,
            selectedMonth,
            viewMode,
            payDay,
          }),
        [expenses, investments, salaryAmount, extraIncomes, selectedMonth, viewMode, payDay]
      );
    }
    ```

    This hook:
    - Takes the same parameters as WaterfallInput but as individual arguments (standard hook ergonomics)
    - Wraps in useMemo with all 7 dependencies in the dependency array for automatic re-computation on any data change (FLOW-04: waterfall updates when user adds expense)
    - Returns WaterfallBar[] -- the same type WaterfallChart expects as `data` prop
    - Does NOT import useMoneyTracker or usePayPeriod itself -- it receives data as parameters (props-only pattern for Phase 21 MonthlyFlowPanel)
  </action>
