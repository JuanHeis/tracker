Add a useMemo:
```typescript
const miniProjectionChartData = useMemo(() => {
  return miniProjection.patrimonyData

      optimista: p.proyeccionOptimista ?? 0,
      base: p.proyeccionBase ?? 0,
      pesimista: p.proyeccionPesimista ?? 0,
```
