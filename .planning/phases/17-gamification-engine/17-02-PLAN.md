       ```typescript
         const scenarios = projectPatrimonyScenarios(currentPatrimony, netSavings, horizonMonths);
         const baseProjection = scenarios.base;
         const simulatedProjection = applySimulatedExpenses(baseProjection, expenses, globalUsdRate);
         const chartData = buildSimulatorData(baseProjection, simulatedProjection, horizonMonths);
         const summary = computeSimulatorSummary(expenses, simulatedProjection, globalUsdRate);
         return { chartData, summary };

       ```
